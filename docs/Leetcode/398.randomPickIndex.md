class Solution:

    def __init__(self, nums: List[int]):
        self.map = {}
        for idx, nums in enumerate(nums):
            if nums not in self.map:
                self.map[nums] = []
            self.map[nums].append(idx)

    def pick(self, target: int) -> int:
        if target in self.map:
            return random.choice(self.map[target])
        else:
            return None
